# 1. Create a payment link (server-side)
import httpx

response = httpx.post(
    f"{API_URL}/api/v1/merchants/{MERCHANT_ID}/payment-links",
    params={"mode": "TEST"},
    headers={"Authorization": f"Bearer {YOUR_SECRET_KEY}"},
    json={
        "title": "My Product",
        "amount": 4200,
        "currency": "usd",
        "redirectUrl": "https://yourshop.com/thank-you",
    },
)
link = response.json()

# 2. Send link["payUrl"] to your customer
print(link["payUrl"])  # https://pay.yourgateway.com/pay/TOKEN
curl -X POST "${API_URL}/api/v1/merchants/${MERCHANT_ID}/payment-links?mode=TEST" \
  -H "Authorization: Bearer ${YOUR_SECRET_KEY}" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "My Product",
    "amount": 4200,
    "currency": "usd",
    "redirectUrl": "https://yourshop.com/thank-you"
  }'
